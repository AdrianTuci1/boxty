=== STEP 1: Register + Login ===

```bash
# Register
REG=$(curl -s -X POST http://localhost:3000/api/auth/register \
  -H "Content-Type: application/json" \
  -d '{"email":"e2e@boxty.dev","password":"test123","name":"E2E Test"}')
echo "Register: $REG"

# Login pentru JWT
LOGIN=$(curl -s -X POST http://localhost:3000/api/auth/login \
  -H "Content-Type: application/json" \
  -d '{"email":"e2e@boxty.dev","password":"test123"}')
TOKEN=$(echo $LOGIN | python3 -c "import sys,json; print(json.load(sys.stdin)['token'])")
echo "✅ Auth — token received: ${TOKEN:0:20}..."
```

=== STEP 2: Create API Key (for SDK tests) ===

```bash
# Creează API key
API_KEY_RESP=$(curl -s -X POST http://localhost:3000/api/auth/api-keys \
  -H "Content-Type: application/json" \

  -d '{"name":"e2e-test-key","permissions":["read","write"]}')

```

=== STEP 5: Forward Port + Test Gateway ===

```bash
# Forward port 8080
FWD=$(curl -s -X POST "http://localhost:3000/api/sandboxes/$SANDBOX_ID/forward" \
  -H "Content-Type: application/json" \
  -H "Authorization: Bearer $TOKEN" \
  -d '{"port":8080}')
echo "Forward response: $FWD"

URL=$(echo $FWD | python3 -c "import sys,json; print(json.load(sys.stdin)['url'])")
echo "Public URL: $URL"

# Extrage hostname
HOSTNAME=$(echo $URL | sed 's|https://||')
echo "Hostname: $HOSTNAME"

# Test through gateway (self-signed TLS)
echo ""
echo "=== GATEWAY TEST ==="
GATEWAY_RESP=$(curl -sk --connect-to "$HOSTNAME:443:localhost:8443" "https://$HOSTNAME/" 2>&1)
echo "Gateway response: $GATEWAY_RESP"

if echo "$GATEWAY_RESP" | grep -q "Hello from Boxty"; then
  echo ""
  echo "✅✅✅ E2E TEST PASSED — Full flow works!"
  echo "  Register → Login → Create Sandbox → Start Server → Forward → Gateway → Response"
else
  echo ""
  echo "❌ Gateway response unexpected"
  echo "  Raw: $GATEWAY_RESP"
fi

# Verify gateway cache
echo ""
echo "--- Gateway cache ---"
curl -s http://localhost:8000/health 2>&1
```
