NU încerca să instalezi Docker sau să pornești dockerd. Verifică doar dacă Docker e deja funcțional.

```bash

docker info 2>&1 | head -3

  echo "❌ Docker nu e disponibil — docker daemon nu rulează"
  echo "   Oprește-te aici și raportează eroarea. NU încerca să instalezi/pornești dockerd."
  exit 1
fi

```

=== STEP 2: Build & Start ===

```bash
cd /tmp/repo/infra/docker
docker compose up -d --build 2>&1
if [ $? -ne 0 ]; then
  echo "❌ docker compose up a eșuat"
  exit 1
fi
echo "✅ Docker Compose pornit"
```

=== STEP 3: Wait for all services healthy ===

```bash
echo "Așteptăm serviciile să fie healthy..."

# API
for i in $(seq 1 30); do
  if curl -s http://localhost:3000/health 2>/dev/null | grep -q '"ok"'; then
    echo "✅ API ready (port 3000)"
    break
  fi
  echo "  waiting API... ($i/30)"
  sleep 2
done

# Gateway
for i in $(seq 1 15); do
  if curl -s http://localhost:8000/health 2>/dev/null | grep -q '"ok"'; then
    echo "✅ Gateway ready (port 8000)"
    break
  fi
  echo "  waiting Gateway... ($i/15)"
  sleep 2
done

# Worker (așteaptă să se înregistreze la API)
for i in $(seq 1 20); do
  WORKERS=$(curl -s http://localhost:3000/api/workers 2>/dev/null)
  if echo "$WORKERS" | grep -q '"active"'; then
    echo "✅ Worker registered"
    break
  fi
  echo "  waiting Worker registration... ($i/20)"
  sleep 3
done
```
