Directorul de lucru: `/tmp/repo`. Nu e nevoie de servicii pornite — testele sunt unitare (mock).

=== STEP 1: Python SDK — install + test ===

```bash
cd /tmp/repo/sdk-py
echo "--- Python SDK install ---"
pip install -e . 2>&1 | tail -5
if [ $? -ne 0 ]; then
  echo "❌ SDK Python install eșuat"
  exit 1
fi
echo "✅ SDK Python installed"

echo ""
echo "--- Python SDK unit tests ---"
python3 -m pytest test/ -v 2>&1 || python3 -m unittest discover test/ -v 2>&1
if [ $? -eq 0 ]; then
  echo "✅ Python SDK tests PASSED"
else
  echo "❌ Python SDK tests FAILED"
fi
```

=== STEP 2: JavaScript SDK — install + build + test ===

```bash
cd /tmp/repo/sdk-js
echo "--- SDK JS install ---"
npm install 2>&1 | tail -5
echo "✅ SDK JS deps installed"

echo ""
echo "--- SDK JS build ---"
npm run build 2>&1
if [ $? -ne 0 ]; then
  echo "❌ SDK JS build eșuat"
  exit 1
fi
echo "✅ SDK JS compiles"

echo ""
echo "--- SDK JS unit tests ---"
# Verifică ce test runner e configurat
if [ -f node_modules/.bin/vitest ]; then
  npx vitest run 2>&1
elif [ -f node_modules/.bin/jest ]; then
  npx jest 2>&1
else
  # Fallback: import check via Node
  node -e "import('./dist/index.js').then(m => { console.log('✅ SDK JS imports OK'); console.log('Exports:', Object.keys(m)); })" 2>&1
  echo "⚠️ No test runner found — import check only"
fi
JS_EXIT=$?
if [ $JS_EXIT -eq 0 ]; then
  echo "✅ JS SDK tests PASSED"
else
  echo "❌ JS SDK tests FAILED"
fi

echo ""
echo "--- SDK JS lint ---"
npm run lint 2>&1 || echo "⚠️ Lint warnings"
```
