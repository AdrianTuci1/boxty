=== STEP 1: Clean up test sandboxes ===

```bash
# Login ca e2e user
LOGIN=$(curl -s -X POST http://localhost:3000/api/auth/login \
  -H "Content-Type: application/json" \
  -d '{"email":"e2e@boxty.dev","password":"test123"}')
TOKEN=$(echo $LOGIN | python3 -c "import sys,json; print(json.load(sys.stdin)['token'])")

# List all sandboxes
SBS=$(curl -s http://localhost:3000/api/sandboxes \
  -H "Authorization: Bearer $TOKEN")
echo "Sandbox-uri active: $(echo $SBS | python3 -c "import sys,json; print(len(json.load(sys.stdin)))")"

# Delete all sandboxes
echo $SBS | python3 -c "
import sys, json, subprocess
sbs = json.load(sys.stdin)
for sb in sbs:
    subprocess.run(['curl','-s','-X','DELETE',
        f'http://localhost:3000/api/sandboxes/{sb[\"id\"]}',
        '-H','Authorization: Bearer $TOKEN'])
    print(f'Deleted: {sb[\"id\"]}')
"
echo "✅ All sandboxes cleaned up"
```

=== STEP 2: Verify Gateway Cache Cleared ===

fi
